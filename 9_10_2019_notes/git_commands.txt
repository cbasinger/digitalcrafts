git status #Tells us which files to push up 

git add -A #Adds all the files to be saved

git commit -m "Description of what changes" #Describes how the files were changed

git push origin master #Saves the files to git

git checkout -b [branch name]: Makes a new branch

git branch: tells you which branch you are currently on

git checkout [branch name]: switch between branches

git pull origin [branch name]: #pull the latest changes from git
